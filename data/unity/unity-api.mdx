# Unity API Reference

## Authentication Methods
All authentication methods on the Unity SDK

### `loginWithEmail`
Logs in a user with email and password..

```cs
using mirrorworld_sdk_unity.Runtime;
using mirrorworld_sdk_unity.Runtime.Models.Request.Authentication;
using mirrorworld_sdk_unity.Runtime.Models.Request.Marketplace;
using mirrorworld_sdk_unity.Runtime.Models.Request.Wallet;
using mirrorworld_sdk_unity.Runtime.Services.Implementations;
using mirrorworld_sdk_unity.Runtime.Services.Interfaces;

public class ApiManager : MonoBehaviour
{
void Start()
    {
        IMirrorWorldClient client = new MirrorWorldClient(mirrorworld_sdk_unity.Runtime.Environment.Staging, EnvironmentVersion.V1,
            "xdK36g2iXgX05oXsCyvlkq6XvDQuhG1mUG2");

StartCoroutine(client.Authentication.LoginWithEmail(
            new BasicEmailLoginRequest()
            {
                Email = "<Emial>",
                Password = "<Password>"
            }, (r =>
            {
                string result = JsonConvert.SerializeObject(r);
                Debug.Log(result);
            })));
    }
}
```
